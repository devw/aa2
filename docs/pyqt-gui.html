Window with text


import sys
from PyQt5.QtWidgets import QApplication, QLabel, QMainWindow
from PyQt5.QtCore import Qt, QCoreApplication

# On fait hériter notre classe de QMainWindow
class MainWindow(QMainWindow):

    def __init__(self):
        super().__init__()

        # Définir un titre pour la fenêtre
        self.setWindowTitle("Ma super fenêtre")
        self.label = QLabel("Voici une fenêtre PyQt5 !")

        # Centrer le texte par rapport à la widget contenante
        self.label.setAlignment(Qt.AlignCenter)

        # Mettre le label dans la central widget
        self.setCentralWidget(self.label)

app = QCoreApplication.instance() # https://qt.developpez.com/doc/4.7/qcoreapplication/#instance
if app is None:
    app = QApplication(sys.argv)

window = MainWindow()
window.show()
app.exec_()

Window with Label and Images


import sys
from PyQt5.QtWidgets import QApplication, QLabel, QMainWindow
from PyQt5.QtCore import Qt, QCoreApplication
from PyQt5.QtGui import QPixmap

class Window(QMainWindow):

    def __init__(self):
    super().__init__()
    self.setWindowTitle("Paint")
    self.label = QLabel('Hello')
    self.label.setPixmap(QPixmap('/tmp/python.png'))
    self.label.setScaledContents(True)
    self.setCentralWidget(self.label)

app = QCoreApplication.instance()
if app is None:
    app = QApplication(sys.argv)

window = Window()
window.show()
app.exec_()
